351  std::string toString() override {
352  try {
353  //return boost::lexical_cast<std::string>(m_val);
354  RWMutexType::ReadLock lock(m_mutex);
355  return ToStr()(m_val);
356  } catch (std::exception &e) {
357  SYLAR_LOG_ERROR(SYLAR_LOG_ROOT()) << "ConfigVar::toString exception "
358  << e.what() << " convert: " << TypeToName<T>() << " to string"
359  << " name=" << m_name;
360  }
361  return "";
362  }
363 
368  bool fromString(const std::string &val) override {
369  try {
370  setValue(FromStr()(val));
371  } catch (std::exception &e) {
372  SYLAR_LOG_ERROR(SYLAR_LOG_ROOT()) << "ConfigVar::fromString exception "
373  << e.what() << " convert: string to " << TypeToName<T>()
374  << " name=" << m_name
375  << " - " << val;
376  }
377  return false;
378  }
379 
383  const T getValue() {
384  RWMutexType::ReadLock lock(m_mutex);
385  return m_val;
386  }
387 
392  void setValue(const T &v) {
393  {
394  RWMutexType::ReadLock lock(m_mutex);
395  if (v == m_val) {
396  return;
397  }
398  for (auto &i : m_cbs) {
399  i.second(m_val, v);
400  }
401  }
402  RWMutexType::WriteLock lock(m_mutex);
403  m_val = v;
404  }
405 
409  std::string getTypeName() const override { return TypeToName<T>(); }
410 
415  uint64_t addListener(on_change_cb cb) {
416  static uint64_t s_fun_id = 0;
417  RWMutexType::WriteLock lock(m_mutex);
418  ++s_fun_id;
419  m_cbs[s_fun_id] = cb;
420  return s_fun_id;
421  }
422 
427  void delListener(uint64_t key) {
428  RWMutexType::WriteLock lock(m_mutex);
429  m_cbs.erase(key);
430  }
431 
437  on_change_cb getListener(uint64_t key) {
438  RWMutexType::ReadLock lock(m_mutex);
439  auto it = m_cbs.find(key);
440  return it == m_cbs.end() ? nullptr : it->second;
441  }
442 
446  void clearListener() {
447  RWMutexType::WriteLock lock(m_mutex);
448  m_cbs.clear();
449  }
450 
451 private:
452  RWMutexType m_mutex;
453  T m_val;
454  //变更回调函数组, uint64_t key,要求唯一，一般可以用hash
455  std::map<uint64_t, on_change_cb> m_cbs;

477  template <class T>
478  static typename ConfigVar<T>::ptr Lookup(const std::string &name,
479  const T &default_value, const std::string &description = "") {
480  RWMutexType::WriteLock lock(GetMutex());
481  auto it = GetDatas().find(name);
482  if (it != GetDatas().end()) {
483  auto tmp = std::dynamic_pointer_cast<ConfigVar<T>>(it->second);
484  if (tmp) {
485  SYLAR_LOG_INFO(SYLAR_LOG_ROOT()) << "Lookup name=" << name << " exists";
486  return tmp;
487  } else {
488  SYLAR_LOG_ERROR(SYLAR_LOG_ROOT()) << "Lookup name=" << name << " exists but type not "
489  << TypeToName<T>() << " real_type=" << it->second->getTypeName()
490  << " " << it->second->toString();
491  return nullptr;
492  }
493  }
494 
495  if (name.find_first_not_of("abcdefghikjlmnopqrstuvwxyz._012345678") != std::string::npos) {
496  SYLAR_LOG_ERROR(SYLAR_LOG_ROOT()) << "Lookup name invalid " << name;
497  throw std::invalid_argument(name);
498  }
499 
500  typename ConfigVar<T>::ptr v(new ConfigVar<T>(name, default_value, description));
501  GetDatas()[name] = v;
502  return v;
503  }
504 
510  template <class T>
511  static typename ConfigVar<T>::ptr Lookup(const std::string &name) {
512  RWMutexType::ReadLock lock(GetMutex());
513  auto it = GetDatas().find(name);
514  if (it == GetDatas().end()) {
515  return nullptr;
516  }
517  return std::dynamic_pointer_cast<ConfigVar<T>>(it->second);
518  }
