  • delListener() : sylar::ConfigVar< T, FromStr, ToStr >
  • delParam() : sylar::http::HttpRequest
  • delServlet() : sylar::http::ServletDispatch
  • Dirname() : sylar::FSUtil
  • dump() : sylar::http::HttpRequest , sylar::http::HttpResponse , sylar::Socket
